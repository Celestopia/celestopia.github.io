CNN手写数字分类

古早练手项目，使用卷积神经网络（CNN）对手写数字进行分类。

数据集：MNIST

              
# 20240214
%%capture --no-display 
#disable unnecessary warnings
%matplotlib inline
import numpy as np
import matplotlib.pyplot as plt
import tensorflow as tf
import keras
              
            
              
(X_train, y_train), (X_test, y_test) = keras.datasets.mnist.load_data()
X_train=X_train.reshape(60000,28,28,1)
X_test=X_test.reshape(10000,28,28,1)

X_train = X_train.astype('float32')
X_test = X_test.astype('float32')
X_train /= 255
X_test /= 255
y_train = keras.utils.to_categorical(y_train, 10)
y_test = keras.utils.to_categorical(y_test, 10)

print('X_train: ',X_train.shape)
print('Y_train: ',y_train.shape)
print('X_test:  ',X_test.shape)
print('Y_test:  ',y_test.shape)
              
            
              
CNN=keras.models.Sequential([
  keras.layers.Conv2D(8, (3, 3), strides=(1, 1), padding='same', input_shape=(28,28,1)),
  keras.layers.LeakyReLU(alpha=0.2),
  #keras.layers.Activation('relu'),
  keras.layers.MaxPooling2D(pool_size=(2, 2)),
  
  keras.layers.Conv2D(16, (3, 3), strides=(1, 1), padding='same'),
  keras.layers.LeakyReLU(alpha=0.2),
  #keras.layers.Activation('relu'),
  keras.layers.MaxPooling2D(pool_size=(2,2)),
  
  keras.layers.Conv2D(32, (3, 3), strides=(1, 1), padding='valid'),
  keras.layers.LeakyReLU(alpha=0.2),
  #keras.layers.Activation('relu'),
  
  keras.layers.Flatten(),
  keras.layers.Dense(64),
  keras.layers.Dense(10),
  keras.layers.Activation('softmax')
])

CNN.summary()
              
            
              
CNN_batch_size=100
CNN_epochs=10
CNN.compile(loss='categorical_crossentropy',optimizer='adam',metrics=['accuracy'])
CNN_history=CNN.fit(
            X_train,y_train,
            batch_size=CNN_batch_size,
            shuffle=True,
            epochs=CNN_epochs,
            verbose=2,
            validation_split=0.2
           )
              
            
              
plt.plot(CNN_history.history['loss'])
plt.plot(CNN_history.history['val_loss'])
plt.title('Model Loss')#图名
plt.ylabel('MSE Loss')#纵坐标名
plt.xlabel('Epoch')#横坐标名
plt.legend(['Train', 'Validation'], loc='upper right')#角标及其位置
plt.show()
              
            
              
plt.plot(CNN_history.history['accuracy'])
plt.plot(CNN_history.history['val_accuracy'])
plt.title('Model accuracy')#图名
plt.ylabel('Accuracy')#纵坐标名
plt.xlabel('Epoch')#横坐标名
plt.legend(['Train', 'Validation'], loc='upper right')#角标及其位置
plt.axhline(y=0.98,c='r')
plt.show()
              
            
              
# Evaluate the model on the test data
print("Evaluate on test data")
results = CNN.evaluate(X_test, y_test, batch_size=20,verbose=2)
print("Test loss", results[0])
print("Test accuracy:", results[1])
